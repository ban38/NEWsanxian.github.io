                            int n;

                            int wo, wt, ho, ht;
                            cin >> wo >> wt >> ho >> ht;

                            int wo_area = wo * wo;
                            int wt_area = wt * wt;

                            vector fill_volumes(n);
                            for (int i = 0; i < n; ++i)
                            {
                                cin >> fill_volumes[i];
                            }

                            double best_height = 0; 

                            for (int volume : fill_volumes)
                            {
                                double current_height = 0; 

                                if (ho > 0)
                                {
                                    
                                    if (volume <= wo_area * ho)
                                    {
                                        current_height = static_cast(volume) / wo_area;
                                        ho -= static_cast(current_height);
                                    }
                                    else
                                    {
                                        
                                        int remaining_volume = volume - wo_area * ho;
                                        current_height = ho; 
                                        ho = 0;

                                        if (remaining_volume <= wt_area * ht)
                                        {
                                            current_height += static_cast(remaining_volume) / wt_area;
                                            ht -= static_cast(remaining_volume) / wt_area;
                                        }
                                        else
                                        {
                                            current_height += ht; 
                                            ht = 0;
                                            break;
                                        }
                                    }
                                }
                                else if (ht > 0)
                                {
                                    
                                    if (volume <= wt_area * ht)
                                    {
                                        current_height = ho + static_cast(volume) / wt_area;
                                        ht -= static_cast(volume) / wt_area;
                                    }
                                    else
                                    {
                                        current_height = ho + ht; 
                                        ht = 0;
                                        break;
                                    }
                                }

                                
                                best_height = max(best_height, current_height);
                            }

                        
                            cout << static_cast(ceil(best_height)) << endl;

                            return 0;
                        }
