# C++ - 解題

1. 輸入一個數字，判斷是否為質數。

這個程式碼會先讓使用者輸入一個數字，然後判斷是否為質數。判斷質數的過程是，先將數字除以2到數字的平方根，如果有餘數，則為非質數。如果沒有餘數，則為質數。

                    #include <iostream>
                    using namespace std;
                    int main() 
                    {
                        int num;
                        cout << "Enter a number: ";
                        cin >> num;
                        bool prime = true;
                        for (int i = 2; i < num; i++) 
                        {
                            if (num % i == 0) 
                            {
                                prime = false;
                                break;
                            }
                        }
                        if (prime) 
                        {
                            cout << num << " is a prime number." << endl;
                        } 
                        else 
                        {
                            cout << num << " is not a prime number." << endl;
                        }
                        return 0;
                    }

兩層容器倒水問題

這個程式碼解決了一個有兩層容器倒水的問題。我們有兩個寬度不同、且有高度限制的容器（上層和下層），每次會有一定的水量倒入這些容器。目標是將水量優先倒入下層容器，若下層滿了才倒入上層，並找出最終倒水結束後上層容器的最大水位高度。

                        #include < iostream >
                        #include < vector >
                        #include <cmath>
                        using namespace std;

                        int main()
                        {
                            int n;
                            cin >> n;

                            int wo, wt, ho, ht;
                            cin >> wo >> wt >> ho >> ht;

                            int wo_area = wo * wo;
                            int wt_area = wt * wt;

                            vector fill_volumes(n);
                            for (int i = 0; i < n; ++i)
                            {
                                cin >> fill_volumes[i];
                            }

                            double best_height = 0; 

                            for (int volume : fill_volumes)
                            {
                                double current_height = 0; 

                                if (ho > 0)
                                {
                                    
                                    if (volume <= wo_area * ho)
                                    {
                                        current_height = static_cast(volume) / wo_area;
                                        ho -= static_cast(current_height);
                                    }
                                    else
                                    {
                                        
                                        int remaining_volume = volume - wo_area * ho;
                                        current_height = ho; 
                                        ho = 0;

                                        if (remaining_volume <= wt_area * ht)
                                        {
                                            current_height += static_cast(remaining_volume) / wt_area;
                                            ht -= static_cast(remaining_volume) / wt_area;
                                        }
                                        else
                                        {
                                            current_height += ht; 
                                            ht = 0;
                                            break;
                                        }
                                    }
                                }
                                else if (ht > 0)
                                {
                                    
                                    if (volume <= wt_area * ht)
                                    {
                                        current_height = ho + static_cast(volume) / wt_area;
                                        ht -= static_cast(volume) / wt_area;
                                    }
                                    else
                                    {
                                        current_height = ho + ht; 
                                        ht = 0;
                                        break;
                                    }
                                }

                                
                                best_height = max(best_height, current_height);
                            }

                        
                            cout << static_cast(ceil(best_height)) << endl;

                            return 0;
                        }
